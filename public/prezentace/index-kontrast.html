// simulation.ts — jedna iterace protokolu
export function step(state: NetworkState): Step {
  const next = clone(state)
  for (const r of next.routers)
    for (const link of neighbors(r)) {
      const cost = table[link.to] + link.metric
      if (cost < (r.table[dest] ?? Infinity)) {
        r.table[dest] = { via: link.to, cost }   // kratší cesta
        changes.push({ router: r.id, dest })
      }
    }
  return { state: next, changes }
}
